// CareConnect classroom prototype Firestore rules.
// These rules are for the course demo only, not final production rules.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{userId} {
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
    }

    match /familyStates/{userId} {
      allow read, update, delete: if request.auth != null && request.auth.uid == userId;
      allow create: if request.auth != null && request.auth.uid == userId;
    }

    // Reserved for Member B/C/D integration.
    // For classroom demo only: any signed-in user can read/write alerts.
    // Later this should be restricted by familyUid, communityId, or assigned staff role.
    match /alerts/{alertId} {
      allow read, write: if request.auth != null;
    }
  }
}
